183 void GameBoard::changeStones(int index)
184 {
185  sort(available.begin(), available.end());
186  it = find_if(available.begin(), available.end(), isEqual(index));
187 
188  while (it != available.end() && it->first == index)

196 @brief Function replaces stones till his own stone is found, then function ends
197 */
198 void GameBoard::replaceStones(int index, int dir)
199 {
200  int next = index - dir;
201  vector<int> *vToSet;

204

238 void GameBoard::checkNextSq(int pos, int dir, int enemy, int me)
239 {
240  int index = pos + dir;
241 
242  if (index >= size*size || isBorder(dir, pos) || index < 0 || grid[index] == me) return;
243  else
244  {
245  if (grid[index] == NONE)
246  { /*SEND DIRECTION AND LAST POSITION SO IF HE PLANTS HE CAN GO FROM AVAILABLE POS TO HIS DISC*/
247  grid[index] = AVAIL;
248  available.push_back(make_pair(index, dir));
249  }
250  else if (grid[index] == AVAIL)
251  {
252  available.push_back(make_pair(index, dir));
253  }
254  else checkNextSq(index, dir, enemy, me);
255  }
256 }
