91  return;

95 
96  int x = (event->x() - 10)/size;
97  int y = (event->y() - 10)/size;
98 
99  int index = y*gb->size + x;
100 
101  if (gb->grid[index] == AVAIL) {
102  gb->placeStone(index);
103  save->addState(gb);
104  gb->setAvailables();
105  repaint();
106 
107  //Check if game is over
108  if (gb->checkEnd()) {
109  msgB.setText(" Konec hry.\t");
110  msgB.exec();

116  gb->placeStone(ai->run(gb, save));
