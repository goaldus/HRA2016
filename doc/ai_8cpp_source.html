83 double AI::evalPos(int index, GameBoard * gb) {
84  int x = index / gb->size;
85  int y = index % gb->size;
86  int max = gb->size - 1;
87 
88  if ((x == 0 && (y == 0 || y == max)) || (x == max && (y == 0 || y == max))) {
89  return 1.5;
90  }
91  else if ((x == 0 && (y == 1 || y == max-1)) || (x == 1 && (y == 0 || y == max)) ||
92  (x == max && (y == 1 || y == max-1)) || (x == max-1 && (y == 0 || y == max))) {
93  return 0.4;
94  }
95  else if (((x == 0 || x == max) && (y >= 2 && y < max - 1)) || ((y == 0 || y == max) && (x >= 2 && x < max - 1))) {
96  return 0.7;
97  }
98  else
99  return 1.0;
100 }

128  }
129

132  delete simgb;
